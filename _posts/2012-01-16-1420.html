#!/usr/bin/env python
import os, sys
import zlib
import base64
filename = sys.argv[1]
def crc(fileName):
  fd = open(fileName,"rb")
  content = fd.readlines()
  fd.close()
  prev = None
  for eachLine in content:
    if not prev:
      prev = zlib.crc32(eachLine)
    else:
      prev = zlib.crc32(eachLine, prev)
  return prev
for eachFile in sys.argv[1:]:
  print base64.b64encode(str(crc(eachFile)))
