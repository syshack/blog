#!/usr/bin/env python
#-*-coding:utf-8-*-
a={"a":1,"b":2,"d":1}
b={"a":2,"b":3,"e":2}
#解法一
c=a
for k in b:
     c[k] = c.get(k, 0) + b[k]
print c
